## 导数
diff(x^3 + 2x^2 - x + 1) = - 1 + 3 * (x)^(2) + 4 * x

## 二阶导数
diff(x^4; x; 2) = 12 * (x)^(2)

## 不定积分
integrate(x^2 * ln(x)) = 1/9 * (x)^(3) * (- 1 + 3 * ln(x)) + C

## 定积分
integrate(sin(x)^2; 0; pi) = 1/2 * pi

## 极限
limit(sin(x)/x; 0) = 1

## 级数
sum(1/n^2; 1; 100) = sum(N**(-2), 1, 100)
