## 一元二次
solve(x^2 + 5x + 6 = 0; x) = (-3, -2)

## 因式分解
factor(x^2 - 4) = - 4 + (x)^(2)

## 展开
expand((x+1)^4) = 1 + (x)^(4) + 4 * x + 4 * (x)^(3) + 6 * (x)^(2)

## 多项式GCD
gcd(x^3 - 1; x^2 - 1) = - 1 + x

## 多变量求解
multisolve([x+y=5, x-y=1]; [x, y]) = [3  2]
